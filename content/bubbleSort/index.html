Bubble Sort

There are many sorting algorithms, but Bubble Sort is the only one I can write without looking it up first, so here we go

def bubble_sort(a): 
    sorted = False
    while not sorted:
        sorted = True
        for i in range(0,len(a)-1):
            if a[i]>a[i+1]:
                tmp = a[i]
                a[i] = a[i+1]
                a[i+1] = tmp
                sorted = False
    return a
